#!/bin/bash

# Check if the file is provided
if [ "$#" -ne 1 ]; then
    echo "Usage: $0 <xml_file>"
    exit 1
fi

# Get the file name
XML_FILE=$1

# Check if the file exists
if [ ! -f "$XML_FILE" ]; then
    echo "Error: File '$XML_FILE' not found!"
    exit 1
fi

# Extract the content inside <tag1> using grep and sed
TAG1_CONTENT=$(grep "<tag1>" "$XML_FILE" | sed -n 's:.*<tag1>\(.*\)</tag1>.*:\1:p')

# Check if content was found
if [ -z "$TAG1_CONTENT" ]; then
    echo "Error: <tag1> not found or content is empty."
    exit 1
fi

# Print the extracted content
echo "$TAG1_CONTENT"
