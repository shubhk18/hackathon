#!/bin/bash

LOCAL_FILE="/path/to/local/file"
REMOTE_HOST="user@remote_host"
REMOTE_PATH="/tmp/$(basename $LOCAL_FILE)"
REMOTE_COMMAND="cat $REMOTE_PATH"

# Copy the file
scp "$LOCAL_FILE" "$REMOTE_HOST:$REMOTE_PATH"

# Run the command
ssh "$REMOTE_HOST" "$REMOTE_COMMAND"

# Optional: Clean up the remote file
ssh "$REMOTE_HOST" "rm -f $REMOTE_PATH"
