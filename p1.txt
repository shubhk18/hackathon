#!/usr/bin/perl
use strict;
use warnings;

# Specify input and output files
my $input_file = 'input.txt';
my $output_file = 'output.txt';

# Open input and output files
open(my $in, '<', $input_file) or die "Cannot open $input_file: $!";
open(my $out, '>', $output_file) or die "Cannot open $output_file: $!";

# Process each line
while (my $line = <$in>) {
    # Check if line has more than 110 characters
    if (length($line) > 110) {
        # Replace 3 characters at offset 110
        substr($line, 110, 3, 'XYZ');  # Replace with 'XYZ' or desired characters
    }
    # Write modified line to output file
    print $out $line;
}

# Close files
close($in);
close($out);

print "Replacement complete. Check output.txt for results.\n";
