my $text = "Abc.....shdhsjjs\nShashank’s\nMi.mid in (‘123’)";
my $value;

# Match 'mid' keyword, followed by 'in' and capture the value in parentheses
if ($text =~ /\bmid\b.*?\bin\s*\(\s*['"](\d+)['"]\s*\)/i) {
    $value = $1;  # Capture group 1 contains the number inside the IN clause
    print "Extracted value: $value\n";
} else {
    print "No match found.\n";
}
