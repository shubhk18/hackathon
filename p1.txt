#!/bin/bash

# File path to the XML file
XML_FILE="example.xml"

# Check if the file exists
if [[ ! -f $XML_FILE ]]; then
    echo "File not found: $XML_FILE"
    exit 1
fi

# Use xmllint to extract <name> elements
result=$(xmllint --xpath "//*[local-name()='name']/text()" "$XML_FILE" 2>/dev/null)

# Check if xmllint succeeded
if [[ $? -ne 0 ]]; then
    echo "Error parsing XML"
    exit 1
fi

# Print the extracted names
echo "Extracted Names: $result"
