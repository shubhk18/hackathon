#!/bin/bash

# Define a Bash variable
my_variable="some_value"

# Execute SQL and capture the result in a variable
result=$(sqlplus -s username/password@database <<EOF
SET PAGESIZE 0 FEEDBACK OFF VERIFY OFF HEADING OFF ECHO OFF
SELECT your_column FROM your_table WHERE your_column = '$my_variable';
EXIT;
EOF
)

# Trim whitespace from the result (optional)
result=$(echo $result | xargs)

# Print the result
echo "Query result: $result"
