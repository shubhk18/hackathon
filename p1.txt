#!/bin/bash

line="This is an example pattern check pattern with Bash scripting pattern example"

# Define the delimiter/pattern
pattern="pattern"

# Replace the pattern with a custom delimiter (e.g., space)
IFS="$pattern" read -ra parts <<< "$line"

# Loop through all parts
for i in "${!parts[@]}"; do
    echo "Part $((i+1)): ${parts[i]}"
done
